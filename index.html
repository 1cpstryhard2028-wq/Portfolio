import time
import random


print("Welcome to the password generator")


time.sleep(2)


print(
   '''
Here you will be able to generate a password
using your character traits.
''')


username = input("Enter your name: ") #ask for name
if len(username) < 3 or len(username) > 100:
   print("Please type a reasonable username with more characters than 3 and less characters than 100")
   username = input("Enter your name again: ")
   while True:  #Verify that the user typed in a valid username
       print("Please enter a valid username")
       username = input("Enter your name again: ")
       if len(username) > 3 or len(username) == 3 or len(username) == 100 or len(username) < 100:
           break


age = input("Enter your age .: ") #ask for age
if int(age) < 10:
   print("Please enter a valid age")
   age= input("Enter your age again: ")
   while True : #Verify that the user typed in a valid age
       print("Please enter a valid age")
       age = input("Enter your age again: ")
       if int(age) > 10 or int(age) == 10:
           break


passion = input("Enter your passion: ") #ask for passion
favorite_sport = input("Enter your favorite sport: ") #ask for favorite sport


def scrambling(word):
   new_length = len(word) // 2
   half_word = word[0:new_length]
   return half_word


passion_scramble =scrambling(passion)
favorite_sport_scramble = scrambling(favorite_sport)
username_scramble = scrambling(username)


scramble_list = [passion_scramble, favorite_sport_scramble, username_scramble]
random_password_list = []


password_scrmbl_counter = 0
while True:
   random.shuffle(scramble_list)
   random_password_list.append("".join(scramble_list))
   password_scrmbl_counter += 1
   if password_scrmbl_counter == 50:
       break




appearance_counter = {}
for password in random_password_list:
   if password in appearance_counter:
       appearance_counter[password] += 1
   else:
       appearance_counter[password] = 1


new_unique_list = list(appearance_counter.keys())
random.shuffle(new_unique_list)
newnew_password = new_unique_list[0]+age
print(f"Your generated password is {newnew_password.title()}")


file = open("passwords.txt", "a")
file.write(newnew_password+"\n")
file = open("passwords.txt", "r")
